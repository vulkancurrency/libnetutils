cmake_minimum_required(VERSION 3.8 FATAL_ERROR)
project(libnetutils LANGUAGES C CXX)

option(LIBNETUTILS_LINK_STATIC "Build the libnetutils as a statically-linked library" OFF)
option(LIBNETUTILS_BUILD_UNITTESTS "Build the libnetutils unittests" OFF)

if (NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Debug)
endif()

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)

set(CMAKE_C_FLAGS "-Wall")
set(CMAKE_C_FLAGS_DEBUG "-g")
set(CMAKE_C_FLAGS_RELEASE "-O3")

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

set(CMAKE_CXX_FLAGS "-Wall")
set(CMAKE_CXX_FLAGS_DEBUG "-g")
set(CMAKE_CXX_FLAGS_RELEASE "-O3")

set(BUILD_UVW_LIBS ON)
set(BUILD_TESTING OFF)
add_subdirectory(external/uvw)
include_directories(external/uvw/src)

add_subdirectory(src)

if (LIBNETUTILS_BUILD_UNITTESTS)
  add_subdirectory(external/googletest)
  include_directories(external/googletest/googletest/include)
  add_subdirectory(tests)
endif()
